import "fmt"

// NON-BLOCKING OPS.

func main() {
	// non-blocking RECEIVE OP.

	ch := make(chan int)

	select {
	case msg:=<-ch:
		fmt.Println("RECEIVED:",msg)
	default:
		fmt.Println("Default: No messages available")
	}

    //ðŸ’¡ OUTPUT: Default: No messages available

	// non-blocking SEND OP.

	case ch<-1:
		fmt.Println("Sent message:")
	default:
		fmt.Println("Default: Channel is not ready to receive")	
	}

}
