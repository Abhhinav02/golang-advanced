```go
package main

// SORTING

import (
	"fmt"
	"sort"
)

func main() {
	nums := []int{5, 3, 4, 2, 1}
	strs:= []string{"Bruce","Peter","Tony","Clark"}

	sort.Ints(nums) // modifies the original slice
	fmt.Println("Sorted numbers:",nums)

	sort.Strings(strs) // modifies the original slice
	fmt.Println("Sorted strings:",strs)

	// OP:
	// Sorted numbers: [1 2 3 4 5]
	// Sorted strings: [Bruce Clark Peter Tony]

}
```
