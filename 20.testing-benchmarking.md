```go
package main

import "testing"

//ðŸ’¡ Table-Driven tests - common pattern for running multiple test cases with varying inputs and expected outputs.

func Add(a,b int)int{
	return a+b
}

func TestAddTableDriven(t *testing.T){
	tests:= []struct{a,b,expected int}{
		{2,3,5},
		{0,0,0},
		{-1,1,0},
	}

	for _,test:=range tests{
		result:=Add(test.a,test.b) // Assertion
		if result != test.expected{
			t.Errorf("Add(%d, %d) = %d; Expected = %d", test.a,test.b, result, test.expected)
		}
	}

}

// O/P:
// $ go test testing_benchmarking_test.go
// ok      command-line-arguments  0.431s
```

```go
package main

import (
	"fmt"
	"testing"
)

//ðŸ’¡ subtests - allow grouping related tests and provides better test-organisation and reporting
// Runs in a different thread/goroutine

func Add(a,b int)int{
	return a+b
}

func TestAddSubtests(t *testing.T){
	tests:= []struct{a,b,expected int}{
		{2,3,5},
		{0,0,0},
		{-1,1,0},
	}

	for _,test:=range tests{
		t.Run(fmt.Sprintf("Add(%d,%d)", test.a, test.b), func(t *testing.T){
			result:=Add(test.a, test.b)
			if result!=test.expected{
				t.Errorf("Result = %d; Expected = %d",result,test.expected)
			}
		})
	}

}

// O/P:
// $ go test testing_benchmarking_test.go
// ok      command-line-arguments  0.343s
```
