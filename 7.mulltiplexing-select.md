### ALT. SYNTAX, RECEIVING MESSAGES
```go
func main() {
	ch := make(chan int)

	go func(){
		ch <-1
		close(ch)
	}()

	for {
		select{
		case msg,ok:= <-ch:
			if !ok{
				fmt.Println("Channel closed üî¥")
				// cleanup activities.. üßπ
				return
			}
			fmt.Println("Received:",msg)
		}
	}
 }
```
```
$ go run .
Received: 1
Channel closed üî¥
```
