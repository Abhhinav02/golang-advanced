```go
package main

import (
	"fmt"
	"time"
)

// Channels - way to communicate between goroutines / threads
// variable := make(chan type)

func main() {
greeting := make(chan string)
greetStr:="Heya!"

go func(){
//! receiving ⬅️
greeting <- greetStr // blocking because it is continuously trynna receive values, it's ready to receive continuous flow of data.
greeting <- "World"

// Another example - receive multiple chan vals through a loop
for _,e:=range "abcdef"{
	greeting <- "\nAlphabet: "+string(e)
}
}()

go func ()  {
//! sending ➡️
receiver := <-greeting
fmt.Println(receiver) // Heya!
receiver = <-greeting
fmt.Println(receiver) // World!

// Send multiple chan vals through a loop
for range 6{ // exact no. of vals received
alphabetRvcr := <-greeting // receiver created
fmt.Println(alphabetRvcr)
}
}()


time.Sleep(1 * time.Second) // avoid memory-leak

fmt.Println("End of program.. ✅") // after 1 second
}
```
