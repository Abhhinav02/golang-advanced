func main() {
	done := make(chan struct{}) // struct{} for a change

	go func() {
		fmt.Println("Working.. ✅")
		time.Sleep(2 * time.Second)
		done<-struct{}{}
	}()

	<- done // unbuffered
	fmt.Println("Finished.. ☑️")
}

func main() {
	ch:=make(chan int)



	go func() {
		fmt.Println("Sending.. ")
		ch<-30 // Blocking until the value is received
		time.Sleep(1 * time.Second)
		fmt.Println("Value sent.. ✅")
	}()

	value:= <- ch // Blocking until a value is send

	fmt.Println("Value:", value)
}
